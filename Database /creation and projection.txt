CREATE TABLE employee (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    salary INT(10) DEFAULT 10000,
    mobile VARCHAR(10) UNIQUE,
    age INT CHECK (age > 17)
);
